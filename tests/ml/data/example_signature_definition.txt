name "small deletion"
1 SVTYPE == DEL
2 SVLEN > -1000000
3 SVLEN < 0
logic 1 & 2 & 3 

name "large deletion"
1 SVTYPE == DEL
2 SVLEN < -1000000
logic 1 & 2

name "small duplication"
1 SVTYPE == DUP
2 SVLEN > -1000000
3 SVLEN < 0
logic 1 & 2 & 3 

name "large duplication"
1 SVTYPE == DUP
2 SVLEN < -1000000
logic 1 & 2

name "small inversion"
1 SVTYPE == t2tINV
2 SVTYPE == h2hINV
3 SVLEN > -1000000
4 SVLEN < 0
logic (1 | 2) & 3 & 4

name "large inversion"
1 SVTYPE == t2tINV
2 SVTYPE == h2hINV
2 SVLEN < -1000000
logic (1 | 2) & 3

name "translocation"
1 SVTYPE == TRA
logic 1
